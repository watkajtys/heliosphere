async function gradeSunDisk(imageBuffer) {
    return await sharp(imageBuffer)
        .modulate({ saturation: 1.3, brightness: 1.05, hue: 15 })
        .tint({ r: 255, g: 200, b: 120 })
        .linear(1.2, -10)
        .gamma(1.0)
        .toBuffer();
}
